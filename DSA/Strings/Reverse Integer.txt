int reverse(int x){
    int digit,rev=0,num;
    if (x == INT_MIN) {
        return 0;
    }

    num=(x<0)?-1:1;
    x=(x<0)?-1*x:x;
    while(x>0)
    {
        if (rev > INT_MAX / 10 || (rev == INT_MAX / 10 && digit > 7)) {
            return 0;  // Return 0 in case of overflow
        }
        digit=x%10;
        rev=(rev*10)+digit;
        x/=10;

    }
rev=rev*num;
return rev;


}
